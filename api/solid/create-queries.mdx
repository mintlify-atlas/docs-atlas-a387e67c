---
title: createQueries
description: Solid primitive for fetching multiple queries in parallel
---

The `createQueries` primitive allows you to fetch multiple queries in parallel while maintaining their individual states.

## Import

```tsx
import { createQueries } from '@tanstack/solid-query'
```

## Signature

```ts
function createQueries<T extends Array<any>, TCombinedResult = QueriesResults<T>>(
  queriesOptions: Accessor<{
    queries: readonly [...QueriesOptions<T>]
    combine?: (result: QueriesResults<T>) => TCombinedResult
  }>,
  queryClient?: Accessor<QueryClient>,
): TCombinedResult
```

## Parameters

<ParamField path="queriesOptions" type="Accessor<object>" required>
  An accessor function that returns configuration for multiple queries.

  <Expandable title="properties">
    <ParamField path="queries" type="QueriesOptions<T>" required>
      An array of query options objects. Each element should be a query configuration.

      ```ts
      queries: [
        { queryKey: ['todo', 1], queryFn: () => fetchTodo(1) },
        { queryKey: ['todo', 2], queryFn: () => fetchTodo(2) },
      ]
      ```
    </ParamField>

    <ParamField path="combine" type="(result: QueriesResults<T>) => TCombinedResult">
      Optional function to combine or transform the results from all queries.

      ```ts
      combine: (results) => ({
        data: results.map(r => r.data),
        pending: results.some(r => r.isPending),
      })
      ```
    </ParamField>
  </Expandable>
</ParamField>

<ParamField path="queryClient" type="Accessor<QueryClient>">
  Custom QueryClient accessor.
</ParamField>

## Returns

Returns a store containing an array of query results, or the combined result if `combine` is used.

<ResponseField name="[n].data" type="TData | undefined">
  The data for the nth query.
</ResponseField>

<ResponseField name="[n].error" type="TError | null">
  The error for the nth query.
</ResponseField>

<ResponseField name="[n].status" type="'pending' | 'error' | 'success'">
  The status of the nth query.
</ResponseField>

<ResponseField name="[n].isPending" type="boolean">
  `true` if the nth query is pending.
</ResponseField>

<ResponseField name="[n].isError" type="boolean">
  `true` if the nth query has errored.
</ResponseField>

<ResponseField name="[n].isSuccess" type="boolean">
  `true` if the nth query has succeeded.
</ResponseField>

## Example

```tsx
import { createQueries } from '@tanstack/solid-query'
import { createSignal, For } from 'solid-js'

function TodosList() {
  const [todoIds, setTodoIds] = createSignal([1, 2, 3])

  const results = createQueries(() => ({
    queries: todoIds().map((id) => ({
      queryKey: ['todo', id],
      queryFn: async () => {
        const response = await fetch(
          `https://api.example.com/todos/${id}`
        )
        return response.json()
      },
    })),
  }))

  return (
    <div>
      <For each={results}>
        {(result, index) => (
          <div>
            <Show when={result.isPending}>Loading...</Show>
            <Show when={result.isError}>
              Error: {result.error?.message}
            </Show>
            <Show when={result.isSuccess}>
              {result.data?.title}
            </Show>
          </div>
        )}
      </For>
    </div>
  )
}
```

## With Combine

```tsx
import { createQueries } from '@tanstack/solid-query'

function UserProfile() {
  const combinedResult = createQueries(() => ({
    queries: [
      { queryKey: ['user', 1], queryFn: () => fetchUser(1) },
      { queryKey: ['posts', 1], queryFn: () => fetchUserPosts(1) },
    ],
    combine: (results) => ({
      user: results[0].data,
      posts: results[1].data,
      isPending: results.some((r) => r.isPending),
      isError: results.some((r) => r.isError),
    }),
  }))

  return (
    <div>
      <Show when={combinedResult.isPending}>Loading...</Show>
      <Show when={combinedResult.isError}>Error occurred</Show>
      <Show when={!combinedResult.isPending && !combinedResult.isError}>
        <h2>{combinedResult.user?.name}</h2>
        <For each={combinedResult.posts}>
          {(post) => <div>{post.title}</div>}
        </For>
      </Show>
    </div>
  )
}
```

## Reactive Queries

```tsx
import { createQueries } from '@tanstack/solid-query'
import { createSignal } from 'solid-js'

function DynamicQueries() {
  const [count, setCount] = createSignal(3)

  const results = createQueries(() => ({
    queries: Array.from({ length: count() }, (_, i) => ({
      queryKey: ['todo', i + 1],
      queryFn: () => fetchTodo(i + 1),
    })),
  }))

  return (
    <div>
      <button onClick={() => setCount(count() + 1)}>
        Add Query
      </button>
      <div>Fetching {count()} todos</div>
    </div>
  )
}
```
