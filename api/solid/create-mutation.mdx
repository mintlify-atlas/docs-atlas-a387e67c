---
title: createMutation
description: Solid primitive for performing mutations with automatic state management
---

The `createMutation` primitive is used for creating, updating, and deleting data. Unlike queries, mutations are not automatically executed and require manual invocation.

## Import

```tsx
import { createMutation } from '@tanstack/solid-query'
```

## Signature

```ts
function createMutation<
  TData = unknown,
  TError = DefaultError,
  TVariables = void,
  TOnMutateResult = unknown,
>(
  options: Accessor<UseMutationOptions<TData, TError, TVariables, TOnMutateResult>>,
  queryClient?: Accessor<QueryClient>,
): UseMutationResult<TData, TError, TVariables, TOnMutateResult>
```

## Type Parameters

- `TData` - The type of data returned by the mutation function
- `TError` - The type of error that can be thrown (defaults to `DefaultError`)
- `TVariables` - The type of variables passed to the mutation function (defaults to `void`)
- `TOnMutateResult` - The type returned by the `onMutate` callback (defaults to `unknown`)

## Parameters

<ParamField path="options" type="Accessor<UseMutationOptions>" required>
  An accessor function that returns the mutation configuration object.

  <Expandable title="properties">
    <ParamField path="mutationFn" type="(variables: TVariables) => Promise<TData>" required>
      The function that performs the mutation. Receives the variables passed to `mutate`.

      ```ts
      mutationFn: async (newTodo) => {
        const response = await fetch('/api/todos', {
          method: 'POST',
          body: JSON.stringify(newTodo),
        })
        return response.json()
      }
      ```
    </ParamField>

    <ParamField path="mutationKey" type="MutationKey">
      Optional unique key for the mutation. Used for mutation caching and deduplication.
    </ParamField>

    <ParamField path="onMutate" type="(variables: TVariables) => Promise<TOnMutateResult> | TOnMutateResult">
      Callback fired before the mutation function is executed. Useful for optimistic updates.
    </ParamField>

    <ParamField path="onSuccess" type="(data: TData, variables: TVariables, context: TOnMutateResult) => Promise<unknown> | unknown">
      Callback fired when the mutation succeeds.
    </ParamField>

    <ParamField path="onError" type="(error: TError, variables: TVariables, context: TOnMutateResult | undefined) => Promise<unknown> | unknown">
      Callback fired when the mutation encounters an error.
    </ParamField>

    <ParamField path="onSettled" type="(data: TData | undefined, error: TError | null, variables: TVariables, context: TOnMutateResult | undefined) => Promise<unknown> | unknown">
      Callback fired when the mutation is either successful or encounters an error.
    </ParamField>

    <ParamField path="retry" type="boolean | number | ((failureCount: number, error: TError) => boolean)" default="0">
      Number of times to retry the mutation on failure.
    </ParamField>
  </Expandable>
</ParamField>

<ParamField path="queryClient" type="Accessor<QueryClient>">
  Custom QueryClient accessor.
</ParamField>

## Returns

Returns a store with reactive mutation state and mutation functions:

<ResponseField name="data" type="TData | undefined">
  The data returned by the mutation function.
</ResponseField>

<ResponseField name="error" type="TError | null">
  The error object if the mutation failed.
</ResponseField>

<ResponseField name="status" type="'idle' | 'pending' | 'error' | 'success'">
  The current status of the mutation.
</ResponseField>

<ResponseField name="isPending" type="boolean">
  `true` if the mutation is currently executing.
</ResponseField>

<ResponseField name="isIdle" type="boolean">
  `true` if the mutation has not been executed yet.
</ResponseField>

<ResponseField name="isError" type="boolean">
  `true` if the mutation has errored.
</ResponseField>

<ResponseField name="isSuccess" type="boolean">
  `true` if the mutation has successfully completed.
</ResponseField>

<ResponseField name="mutate" type="(variables: TVariables, options?: MutateOptions) => void">
  Function to trigger the mutation. This is a synchronous function that doesn't return a promise.
</ResponseField>

<ResponseField name="mutateAsync" type="(variables: TVariables, options?: MutateOptions) => Promise<TData>">
  Function to trigger the mutation that returns a promise.
</ResponseField>

## Example

```tsx
import { createMutation, useQueryClient } from '@tanstack/solid-query'
import { createSignal } from 'solid-js'

function TodoForm() {
  const queryClient = useQueryClient()
  const [title, setTitle] = createSignal('')

  const mutation = createMutation(() => ({
    mutationFn: async (newTodo: { title: string; completed: boolean }) => {
      const response = await fetch('https://api.example.com/todos', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(newTodo),
      })
      if (!response.ok) throw new Error('Failed to create todo')
      return response.json()
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['todos'] })
      setTitle('')
    },
  }))

  const handleSubmit = (e: Event) => {
    e.preventDefault()
    mutation.mutate({ title: title(), completed: false })
  }

  return (
    <form onSubmit={handleSubmit}>
      <input
        value={title()}
        onInput={(e) => setTitle(e.currentTarget.value)}
        placeholder="Enter todo title"
      />
      <button type="submit" disabled={mutation.isPending}>
        {mutation.isPending ? 'Creating...' : 'Create Todo'}
      </button>
      <Show when={mutation.isError}>
        <div>Error: {mutation.error?.message}</div>
      </Show>
    </form>
  )
}
```

## Optimistic Updates

```tsx
import { createMutation, useQueryClient } from '@tanstack/solid-query'

function TodoList() {
  const queryClient = useQueryClient()

  const mutation = createMutation(() => ({
    mutationFn: updateTodo,
    onMutate: async (newTodo) => {
      await queryClient.cancelQueries({ queryKey: ['todos'] })
      const previousTodos = queryClient.getQueryData(['todos'])
      
      queryClient.setQueryData(['todos'], (old) => {
        return old.map((todo) =>
          todo.id === newTodo.id ? newTodo : todo
        )
      })
      
      return { previousTodos }
    },
    onError: (err, newTodo, context) => {
      queryClient.setQueryData(['todos'], context.previousTodos)
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['todos'] })
    },
  }))

  return (
    <div>
      {/* Todo list UI */}
    </div>
  )
}
```
