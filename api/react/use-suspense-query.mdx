---
title: useSuspenseQuery
description: React hook for fetching data with React Suspense integration
---

The `useSuspenseQuery` hook is a version of `useQuery` that integrates with React Suspense. It suspends rendering while the query is loading and always returns data (never undefined).

## Import

```tsx
import { useSuspenseQuery } from '@tanstack/react-query'
```

## Signature

```tsx
function useSuspenseQuery<
  TQueryFnData = unknown,
  TError = DefaultError,
  TData = TQueryFnData,
  TQueryKey extends QueryKey = QueryKey,
>(
  options: UseSuspenseQueryOptions<TQueryFnData, TError, TData, TQueryKey>,
  queryClient?: QueryClient,
): UseSuspenseQueryResult<TData, TError>
```

## Type Parameters

- `TQueryFnData` - The type of data returned by the query function
- `TError` - The type of error that can be thrown (defaults to `DefaultError`)
- `TData` - The type of data returned by the select function (defaults to `TQueryFnData`)
- `TQueryKey` - The type of the query key (defaults to `QueryKey`)

## Parameters

### options

Type: `UseSuspenseQueryOptions<TQueryFnData, TError, TData, TQueryKey>`

Accepts all `useQuery` options except:
- `enabled` - Always `true` for suspense queries
- `throwOnError` - Always `true` for suspense queries (errors are thrown to error boundaries)
- `placeholderData` - Not supported with suspense queries

The following options are required:

#### queryKey

- **Type:** `TQueryKey`
- **Required:** Yes

A unique identifier for the query.

#### queryFn

- **Type:** `QueryFunction<TQueryFnData, TQueryKey>`
- **Required:** Yes

The function that the query will use to request data. **Note:** `skipToken` is not allowed with `useSuspenseQuery`.

### queryClient

- **Type:** `QueryClient`
- **Optional**

A custom QueryClient instance.

## Returns

Type: `UseSuspenseQueryResult<TData, TError>`

The suspense query result object, which is similar to `UseQueryResult` but with these key differences:

### data

- **Type:** `TData` (never `undefined`)

The data is **always defined** when the component renders (the query suspends until data is available).

### Other Properties

All other properties from `UseQueryResult` are available except:
- `isPlaceholderData` - Not available (suspense queries don't support placeholder data)
- `promise` - Not available in the result type

The result will never have `isPending: true` since the component suspends during the pending state.

## Examples

### Basic Suspense Query

```tsx
import { Suspense } from 'react'
import { useSuspenseQuery } from '@tanstack/react-query'

function Post({ postId }: { postId: number }) {
  const { data } = useSuspenseQuery({
    queryKey: ['post', postId],
    queryFn: async () => {
      const response = await fetch(`/api/posts/${postId}`)
      return response.json()
    },
  })

  // data is always defined - no need to check for undefined!
  return <div>{data.title}</div>
}

function App() {
  return (
    <Suspense fallback={<div>Loading post...</div>}>
      <Post postId={1} />
    </Suspense>
  )
}
```

### Multiple Suspense Queries

```tsx
import { Suspense } from 'react'
import { useSuspenseQuery } from '@tanstack/react-query'

function UserProfile({ userId }: { userId: number }) {
  const { data: user } = useSuspenseQuery({
    queryKey: ['user', userId],
    queryFn: () => fetchUser(userId),
  })

  const { data: posts } = useSuspenseQuery({
    queryKey: ['posts', userId],
    queryFn: () => fetchUserPosts(userId),
  })

  // Both user and posts are always defined
  return (
    <div>
      <h1>{user.name}</h1>
      <ul>
        {posts.map((post) => (
          <li key={post.id}>{post.title}</li>
        ))}
      </ul>
    </div>
  )
}

function App() {
  return (
    <Suspense fallback={<div>Loading profile...</div>}>
      <UserProfile userId={1} />
    </Suspense>
  )
}
```

### Nested Suspense Boundaries

```tsx
import { Suspense } from 'react'
import { useSuspenseQuery } from '@tanstack/react-query'

function Posts({ userId }: { userId: number }) {
  const { data } = useSuspenseQuery({
    queryKey: ['posts', userId],
    queryFn: () => fetchUserPosts(userId),
  })

  return (
    <ul>
      {data.map((post) => (
        <li key={post.id}>{post.title}</li>
      ))}
    </ul>
  )
}

function User({ userId }: { userId: number }) {
  const { data } = useSuspenseQuery({
    queryKey: ['user', userId],
    queryFn: () => fetchUser(userId),
  })

  return (
    <div>
      <h1>{data.name}</h1>
      <Suspense fallback={<div>Loading posts...</div>}>
        <Posts userId={userId} />
      </Suspense>
    </div>
  )
}

function App() {
  return (
    <Suspense fallback={<div>Loading user...</div>}>
      <User userId={1} />
    </Suspense>
  )
}
```

### With Error Boundary

```tsx
import { Suspense } from 'react'
import { ErrorBoundary } from 'react-error-boundary'
import { useSuspenseQuery } from '@tanstack/react-query'

function Post({ postId }: { postId: number }) {
  const { data } = useSuspenseQuery({
    queryKey: ['post', postId],
    queryFn: async () => {
      const response = await fetch(`/api/posts/${postId}`)
      if (!response.ok) throw new Error('Failed to fetch post')
      return response.json()
    },
  })

  return <div>{data.title}</div>
}

function App() {
  return (
    <ErrorBoundary fallback={<div>Error loading post</div>}>
      <Suspense fallback={<div>Loading...</div>}>
        <Post postId={1} />
      </Suspense>
    </ErrorBoundary>
  )
}
```

### With Select

```tsx
function TodoNames() {
  const { data } = useSuspenseQuery({
    queryKey: ['todos'],
    queryFn: fetchTodos,
    select: (todos) => todos.map((todo) => todo.name),
  })

  // data is always defined as string[]
  return <div>{data.join(', ')}</div>
}
```

### Refetching

```tsx
function Post({ postId }: { postId: number }) {
  const { data, refetch, isRefetching } = useSuspenseQuery({
    queryKey: ['post', postId],
    queryFn: () => fetchPost(postId),
  })

  return (
    <div>
      <h1>{data.title}</h1>
      <button onClick={() => refetch()} disabled={isRefetching}>
        {isRefetching ? 'Refreshing...' : 'Refresh'}
      </button>
    </div>
  )
}
```

### Type Safety

```tsx
type Post = {
  id: number
  title: string
  body: string
}

function Post({ postId }: { postId: number }) {
  const { data } = useSuspenseQuery<Post>({
    queryKey: ['post', postId],
    queryFn: async () => {
      const response = await fetch(`/api/posts/${postId}`)
      return response.json()
    },
  })

  // data is always typed as Post (never undefined)
  return (
    <div>
      <h1>{data.title}</h1>
      <p>{data.body}</p>
    </div>
  )
}
```
