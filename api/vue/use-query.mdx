---
title: useQuery
description: Vue composable for fetching and caching data with automatic reactivity
---

The `useQuery` composable is the primary way to fetch and cache data in Vue Query. It subscribes to a query and returns reactive state.

## Import

```vue
import { useQuery } from '@tanstack/vue-query'
```

## Signature

```ts
function useQuery<
  TQueryFnData = unknown,
  TError = DefaultError,
  TData = TQueryFnData,
  TQueryKey extends QueryKey = QueryKey,
>(
  options: MaybeRefOrGetter<UseQueryOptions<TQueryFnData, TError, TData, TQueryFnData, TQueryKey>>,
  queryClient?: QueryClient,
): UseQueryReturnType<TData, TError>
```

## Type Parameters

- `TQueryFnData` - The type of data returned by the query function
- `TError` - The type of error that can be thrown (defaults to `DefaultError`)
- `TData` - The type of data returned by the select function (defaults to `TQueryFnData`)
- `TQueryKey` - The type of the query key (defaults to `QueryKey`)

## Parameters

<ParamField path="options" type="MaybeRefOrGetter<UseQueryOptions>" required>
  Query configuration object. Can be a ref, getter function, or plain object. All properties can be reactive using Vue refs.

  <Expandable title="properties">
    <ParamField path="queryKey" type="TQueryKey" required>
      A unique identifier for the query. The query will automatically update when this key changes.

      ```vue
      queryKey: ['todos']
      queryKey: ['todos', todoId]
      queryKey: ['todos', ref(filter)]
      ```
    </ParamField>

    <ParamField path="queryFn" type="QueryFunction<TQueryFnData, TQueryKey>" required>
      The function that the query will use to request data. Receives a `QueryFunctionContext` object.

      ```ts
      queryFn: async ({ queryKey, signal }) => {
        const [, todoId] = queryKey
        const response = await fetch(`/api/todos/${todoId}`, { signal })
        return response.json()
      }
      ```
    </ParamField>

    <ParamField path="enabled" type="MaybeRefOrGetter<boolean>" default="true">
      Set to `false` to disable the query from automatically running. Can be a ref or getter function for reactive enabling.
    </ParamField>

    <ParamField path="staleTime" type="number | ((query: Query) => number)" default="0">
      The time in milliseconds after data is considered stale.
    </ParamField>

    <ParamField path="gcTime" type="number | Infinity" default="300000">
      The time in milliseconds that unused/inactive cache data remains in memory (5 minutes by default).
    </ParamField>

    <ParamField path="retry" type="boolean | number | ((failureCount: number, error: TError) => boolean)" default="3">
      Configure retry behavior for failed queries.
    </ParamField>

    <ParamField path="refetchInterval" type="number | false | ((query: Query) => number | false)" default="false">
      If set to a number, the query will continuously refetch at this frequency in milliseconds.
    </ParamField>

    <ParamField path="refetchOnMount" type="boolean | 'always'" default="true">
      If set to `true`, the query will refetch on mount if the data is stale.
    </ParamField>

    <ParamField path="refetchOnWindowFocus" type="boolean | 'always'" default="true">
      If set to `true`, the query will refetch on window focus if the data is stale.
    </ParamField>

    <ParamField path="refetchOnReconnect" type="boolean | 'always'" default="true">
      If set to `true`, the query will refetch on reconnect if the data is stale.
    </ParamField>

    <ParamField path="select" type="(data: TQueryFnData) => TData">
      Transform or select a part of the data returned by the query function.
    </ParamField>

    <ParamField path="initialData" type="TQueryFnData | () => TQueryFnData">
      Initial data to use for the query. When provided, the query will start in a success state.
    </ParamField>

    <ParamField path="placeholderData" type="TQueryFnData | () => TQueryFnData">
      Placeholder data to use while the query is loading. Unlike `initialData`, this doesn't affect the cache.
    </ParamField>
  </Expandable>
</ParamField>

<ParamField path="queryClient" type="QueryClient">
  Custom QueryClient instance. If not provided, the one from the nearest context will be used.
</ParamField>

## Returns

Returns reactive refs with the query state:

<ResponseField name="data" type="Ref<TData | undefined>">
  The data returned by the query function.
</ResponseField>

<ResponseField name="error" type="Ref<TError | null>">
  The error object if the query failed.
</ResponseField>

<ResponseField name="status" type="Ref<'pending' | 'error' | 'success'>">
  The current status of the query.
</ResponseField>

<ResponseField name="isPending" type="Ref<boolean>">
  `true` if the query is in a pending state (initial load with no data).
</ResponseField>

<ResponseField name="isLoading" type="Ref<boolean>">
  `true` if the query is fetching for the first time.
</ResponseField>

<ResponseField name="isError" type="Ref<boolean>">
  `true` if the query has errored.
</ResponseField>

<ResponseField name="isSuccess" type="Ref<boolean>">
  `true` if the query has successfully loaded data.
</ResponseField>

<ResponseField name="isFetching" type="Ref<boolean>">
  `true` if the query is currently fetching (includes background refetches).
</ResponseField>

<ResponseField name="refetch" type="() => Promise<UseQueryResult>">
  Function to manually refetch the query.
</ResponseField>

## Example

```vue
<script setup>
import { useQuery } from '@tanstack/vue-query'
import { ref } from 'vue'

const todoId = ref(1)

const { data, isPending, isError, error } = useQuery({
  queryKey: ['todo', todoId],
  queryFn: async () => {
    const response = await fetch(`https://api.example.com/todos/${todoId.value}`)
    if (!response.ok) throw new Error('Failed to fetch todo')
    return response.json()
  },
  staleTime: 5000,
})
</script>

<template>
  <div>
    <div v-if="isPending">Loading...</div>
    <div v-else-if="isError">Error: {{ error.message }}</div>
    <div v-else>{{ data.title }}</div>
  </div>
</template>
```

## Reactive Options

All options can be reactive using Vue refs or getter functions:

```vue
<script setup>
import { useQuery } from '@tanstack/vue-query'
import { ref, computed } from 'vue'

const enabled = ref(true)
const todoId = ref(1)

const { data } = useQuery({
  queryKey: ['todo', todoId],
  queryFn: () => fetchTodo(todoId.value),
  enabled, // Can be a ref
  staleTime: () => enabled.value ? 5000 : 0, // Or a getter function
})
</script>
```
