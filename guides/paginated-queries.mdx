Paginated queries are perfect for traditional page-based navigation. TanStack Query provides the `placeholderData` option with `keepPreviousData` to create a smooth pagination experience.

## Basic Pagination

Here's a real example from the TanStack Query source:

```tsx
import React from 'react'
import {
  useQuery,
  useQueryClient,
  keepPreviousData,
} from '@tanstack/react-query'

type ProjectsResponse = {
  projects: Array<{ name: string; id: number }>
  hasMore: boolean
}

const fetchProjects = async (page = 0): Promise<ProjectsResponse> => {
  const response = await fetch(`/api/projects?page=${page}`)
  return await response.json()
}

function Projects() {
  const queryClient = useQueryClient()
  const [page, setPage] = React.useState(0)

  const { status, data, error, isFetching, isPlaceholderData } = useQuery({
    queryKey: ['projects', page],
    queryFn: () => fetchProjects(page),
    placeholderData: keepPreviousData,
    staleTime: 5000,
  })

  // Prefetch the next page!
  React.useEffect(() => {
    if (!isPlaceholderData && data?.hasMore) {
      queryClient.prefetchQuery({
        queryKey: ['projects', page + 1],
        queryFn: () => fetchProjects(page + 1),
      })
    }
  }, [data, isPlaceholderData, page, queryClient])

  return (
    <div>
      {status === 'pending' ? (
        <div>Loading...</div>
      ) : status === 'error' ? (
        <div>Error: {error.message}</div>
      ) : (
        <div>
          {data.projects.map((project) => (
            <p key={project.id}>{project.name}</p>
          ))}
        </div>
      )}
      <div>Current Page: {page + 1}</div>
      <button
        onClick={() => setPage((old) => Math.max(old - 1, 0))}
        disabled={page === 0}
      >
        Previous Page
      </button>{' '}
      <button
        onClick={() => {
          setPage((old) => (data?.hasMore ? old + 1 : old))
        }}
        disabled={isPlaceholderData || !data?.hasMore}
      >
        Next Page
      </button>
      {isFetching ? <span> Loading...</span> : null}
    </div>
  )
}
```

## Understanding keepPreviousData

The `keepPreviousData` function maintains the previous page's data while fetching the next page:

```tsx
import { useQuery, keepPreviousData } from '@tanstack/react-query'

function Posts({ page }: { page: number }) {
  const { data, isPlaceholderData } = useQuery({
    queryKey: ['posts', page],
    queryFn: () => fetchPosts(page),
    placeholderData: keepPreviousData,
  })

  return (
    <div>
      {data?.posts.map((post) => (
        <article key={post.id}>
          <h2>{post.title}</h2>
        </article>
      ))}
      {isPlaceholderData && <div className="overlay">Loading...</div>}
    </div>
  )
}
```

**What happens:**
1. User clicks "Next Page"
2. Previous page data remains visible (no loading spinner replacing content)
3. New page fetches in the background
4. `isPlaceholderData` is `true` during fetch
5. Data swaps seamlessly when ready

## Page-Based Pagination

### With Page Numbers

```tsx
import React from 'react'
import { useQuery, keepPreviousData } from '@tanstack/react-query'

type PageData = {
  items: Array<{ id: string; name: string }>
  totalPages: number
  currentPage: number
}

function ItemList() {
  const [page, setPage] = React.useState(1)

  const { data, isLoading, isPlaceholderData } = useQuery({
    queryKey: ['items', page],
    queryFn: async (): Promise<PageData> => {
      const res = await fetch(`/api/items?page=${page}`)
      return res.json()
    },
    placeholderData: keepPreviousData,
  })

  if (isLoading) return <div>Loading initial page...</div>

  return (
    <div>
      <div style={{ opacity: isPlaceholderData ? 0.5 : 1 }}>
        {data?.items.map((item) => (
          <div key={item.id}>{item.name}</div>
        ))}
      </div>

      <div className="pagination">
        <button
          onClick={() => setPage((p) => Math.max(1, p - 1))}
          disabled={page === 1}
        >
          Previous
        </button>

        <span>
          Page {page} of {data?.totalPages}
        </span>

        <button
          onClick={() => setPage((p) => p + 1)}
          disabled={page === data?.totalPages}
        >
          Next
        </button>
      </div>
    </div>
  )
}
```

### With Limit and Offset

```tsx
import React from 'react'
import { useQuery, keepPreviousData } from '@tanstack/react-query'

function ProductList() {
  const [offset, setOffset] = React.useState(0)
  const limit = 20

  const { data, isPlaceholderData } = useQuery({
    queryKey: ['products', offset, limit],
    queryFn: async () => {
      const res = await fetch(`/api/products?offset=${offset}&limit=${limit}`)
      return res.json()
    },
    placeholderData: keepPreviousData,
  })

  return (
    <div>
      <div style={{ opacity: isPlaceholderData ? 0.5 : 1 }}>
        {data?.products.map((product) => (
          <ProductCard key={product.id} product={product} />
        ))}
      </div>

      <button
        onClick={() => setOffset((o) => Math.max(0, o - limit))}
        disabled={offset === 0}
      >
        Previous
      </button>

      <button
        onClick={() => setOffset((o) => o + limit)}
        disabled={!data?.hasMore}
      >
        Next
      </button>
    </div>
  )
}
```

## Prefetching Next Page

Prefetch the next page for instant navigation:

```tsx
import React from 'react'
import { useQuery, useQueryClient, keepPreviousData } from '@tanstack/react-query'

function Articles() {
  const queryClient = useQueryClient()
  const [page, setPage] = React.useState(1)

  const { data, isPlaceholderData } = useQuery({
    queryKey: ['articles', page],
    queryFn: () => fetchArticles(page),
    placeholderData: keepPreviousData,
  })

  // Prefetch next page
  React.useEffect(() => {
    if (!isPlaceholderData && data?.hasNextPage) {
      queryClient.prefetchQuery({
        queryKey: ['articles', page + 1],
        queryFn: () => fetchArticles(page + 1),
      })
    }
  }, [data, isPlaceholderData, page, queryClient])

  return (
    <div>
      {data?.articles.map((article) => (
        <ArticlePreview key={article.id} article={article} />
      ))}

      <button
        onClick={() => setPage((p) => p - 1)}
        disabled={page === 1}
      >
        Previous
      </button>

      <button
        onClick={() => setPage((p) => p + 1)}
        disabled={isPlaceholderData || !data?.hasNextPage}
      >
        Next
      </button>
    </div>
  )
}
```

<Warning>
  **Important**: Disable the "Next" button when `isPlaceholderData` is true to prevent users from advancing pages too quickly while data is still loading.
</Warning>

## Page Number Buttons

Implement numbered pagination buttons:

```tsx
import React from 'react'
import { useQuery, useQueryClient, keepPreviousData } from '@tanstack/react-query'

function PaginatedList() {
  const queryClient = useQueryClient()
  const [page, setPage] = React.useState(1)

  const { data, isLoading } = useQuery({
    queryKey: ['items', page],
    queryFn: () => fetchPage(page),
    placeholderData: keepPreviousData,
  })

  if (isLoading) return <div>Loading...</div>

  const totalPages = data?.totalPages ?? 1
  const pages = Array.from({ length: totalPages }, (_, i) => i + 1)

  return (
    <div>
      <ItemList items={data?.items} />

      <div className="pagination">
        {pages.map((pageNum) => (
          <button
            key={pageNum}
            onClick={() => setPage(pageNum)}
            onMouseEnter={() => {
              // Prefetch on hover
              queryClient.prefetchQuery({
                queryKey: ['items', pageNum],
                queryFn: () => fetchPage(pageNum),
              })
            }}
            className={page === pageNum ? 'active' : ''}
          >
            {pageNum}
          </button>
        ))}
      </div>
    </div>
  )
}
```

## Search with Pagination

Combine search and pagination:

```tsx
import React from 'react'
import { useQuery, keepPreviousData } from '@tanstack/react-query'

function SearchResults() {
  const [search, setSearch] = React.useState('')
  const [page, setPage] = React.useState(1)

  const { data, isPlaceholderData } = useQuery({
    queryKey: ['search', search, page],
    queryFn: () => searchItems({ query: search, page }),
    placeholderData: keepPreviousData,
    enabled: search.length > 0,
  })

  // Reset to page 1 when search changes
  React.useEffect(() => {
    setPage(1)
  }, [search])

  return (
    <div>
      <input
        value={search}
        onChange={(e) => setSearch(e.target.value)}
        placeholder="Search..."
      />

      {data && (
        <>
          <div style={{ opacity: isPlaceholderData ? 0.5 : 1 }}>
            {data.results.map((item) => (
              <div key={item.id}>{item.name}</div>
            ))}
          </div>

          <div>
            <button
              onClick={() => setPage((p) => p - 1)}
              disabled={page === 1}
            >
              Previous
            </button>
            <span>Page {page}</span>
            <button
              onClick={() => setPage((p) => p + 1)}
              disabled={!data.hasMore}
            >
              Next
            </button>
          </div>
        </>
      )}
    </div>
  )
}
```

## Loading States

### Differentiate Initial vs Background Loading

```tsx
import { useQuery, keepPreviousData } from '@tanstack/react-query'

function Posts({ page }: { page: number }) {
  const { data, isLoading, isFetching, isPlaceholderData } = useQuery({
    queryKey: ['posts', page],
    queryFn: () => fetchPosts(page),
    placeholderData: keepPreviousData,
  })

  // isLoading: true only on initial load
  if (isLoading) {
    return <FullPageSpinner />
  }

  return (
    <div>
      {/* Dim content while fetching next page */}
      <div style={{ opacity: isPlaceholderData ? 0.6 : 1 }}>
        {data?.posts.map((post) => (
          <PostCard key={post.id} post={post} />
        ))}
      </div>

      {/* Show background loading indicator */}
      {isFetching && !isPlaceholderData && (
        <div>Refreshing...</div>
      )}
    </div>
  )
}
```

## Server-Side Pagination Info

### Total Count Pattern

```tsx
import { useQuery, keepPreviousData } from '@tanstack/react-query'

type PageResponse = {
  items: Item[]
  total: number
  page: number
  pageSize: number
}

function PaginatedItems({ pageSize = 10 }: { pageSize?: number }) {
  const [page, setPage] = React.useState(1)

  const { data } = useQuery({
    queryKey: ['items', page, pageSize],
    queryFn: async (): Promise<PageResponse> => {
      const res = await fetch(`/api/items?page=${page}&size=${pageSize}`)
      return res.json()
    },
    placeholderData: keepPreviousData,
  })

  const totalPages = data ? Math.ceil(data.total / data.pageSize) : 0

  return (
    <div>
      <ItemList items={data?.items} />
      <div>
        Showing {((page - 1) * pageSize) + 1} to {Math.min(page * pageSize, data?.total ?? 0)} of {data?.total} items
      </div>
      <Pagination
        current={page}
        total={totalPages}
        onChange={setPage}
      />
    </div>
  )
}
```

## Pagination with URL State

Sync pagination with URL params:

```tsx
import { useQuery, keepPreviousData } from '@tanstack/react-query'
import { useSearchParams } from 'react-router-dom'

function Products() {
  const [searchParams, setSearchParams] = useSearchParams()
  const page = parseInt(searchParams.get('page') || '1', 10)

  const { data, isPlaceholderData } = useQuery({
    queryKey: ['products', page],
    queryFn: () => fetchProducts(page),
    placeholderData: keepPreviousData,
  })

  const setPage = (newPage: number) => {
    setSearchParams({ page: String(newPage) })
  }

  return (
    <div>
      <ProductList products={data?.products} />
      <button onClick={() => setPage(page - 1)} disabled={page === 1}>
        Previous
      </button>
      <button onClick={() => setPage(page + 1)} disabled={!data?.hasMore}>
        Next
      </button>
    </div>
  )
}
```

## When to Use Pagination vs Infinite Queries

**Use Pagination when:**
- Users need to jump to specific pages
- You need to show page numbers
- Total count is important
- Data has natural page boundaries (e.g., search results)

**Use Infinite Queries when:**
- Implementing "Load More" or infinite scroll
- Mobile-first experiences
- Feeds or timelines
- Total count is not important

<Warning>
  **Performance Tip**: Always use `placeholderData: keepPreviousData` for pagination. Without it, users see a loading spinner on every page change, creating a jarring experience. With it, the previous page remains visible while the next page loads in the background.
</Warning>
