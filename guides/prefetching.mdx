title: Prefetching

Prefetching allows you to load data before it's needed, creating instant, snappy user experiences. TanStack Query provides several methods to prefetch data.

## Basic Prefetching

Use `prefetchQuery` to load data in advance:

```tsx
import { useQueryClient } from '@tanstack/react-query'

function PostList({ posts }: { posts: Post[] }) {
  const queryClient = useQueryClient()

  return (
    <ul>
      {posts.map((post) => (
        <li
          key={post.id}
          onMouseEnter={() => {
            // Prefetch post details on hover
            queryClient.prefetchQuery({
              queryKey: ['post', post.id],
              queryFn: () => fetchPost(post.id),
            })
          }}
        >
          <a href={`/posts/${post.id}`}>{post.title}</a>
        </li>
      ))}
    </ul>
  )
}
```

When the user hovers over a post, its details are prefetched. If they click, the data loads instantly.

## Real Example from Source

Here's a real prefetching example from TanStack Query:

```tsx
import React from 'react'
import { useQuery, useQueryClient } from '@tanstack/react-query'

type Character = {
  id: number
  name: string
}

const getCharacters = async (): Promise<{
  results: Array<Character>
}> => {
  await new Promise((r) => setTimeout(r, 500))
  const response = await fetch('https://rickandmortyapi.com/api/character/')
  return await response.json()
}

const getCharacter = async (selectedChar: number) => {
  await new Promise((r) => setTimeout(r, 500))
  const response = await fetch(
    `https://rickandmortyapi.com/api/character/${selectedChar}`
  )
  return await response.json()
}

function Example() {
  const queryClient = useQueryClient()
  const rerender = React.useState(0)[1]
  const [selectedChar, setSelectedChar] = React.useState(1)

  const charactersQuery = useQuery({
    queryKey: ['characters'],
    queryFn: getCharacters,
  })

  const characterQuery = useQuery({
    queryKey: ['character', selectedChar],
    queryFn: () => getCharacter(selectedChar),
  })

  return (
    <div>
      <p>
        Hovering over a character will prefetch it, and when it's been
        prefetched it will turn <strong>bold</strong>. Clicking on a prefetched
        character will show their stats below immediately.
      </p>
      <h2>Characters</h2>
      {charactersQuery.isPending ? (
        'Loading...'
      ) : (
        <ul>
          {charactersQuery.data?.results.map((char) => (
            <li
              key={char.id}
              onClick={() => setSelectedChar(char.id)}
              onMouseEnter={async () => {
                // Prefetch character details
                await queryClient.prefetchQuery({
                  queryKey: ['character', char.id],
                  queryFn: () => getCharacter(char.id),
                  staleTime: 10 * 1000, // only prefetch if older than 10 seconds
                })

                setTimeout(() => {
                  rerender({})
                }, 1)
              }}
            >
              <div
                style={
                  // Show bold if already cached
                  queryClient.getQueryData(['character', char.id])
                    ? { fontWeight: 'bold' }
                    : {}
                }
              >
                {char.id} - {char.name}
              </div>
            </li>
          ))}
        </ul>
      )}

      <h3>Selected Character</h3>
      {characterQuery.isPending ? (
        'Loading...'
      ) : (
        <pre>{JSON.stringify(characterQuery.data, null, 2)}</pre>
      )}
    </div>
  )
}
```

## Prefetching Methods

### queryClient.prefetchQuery

Prefetch a query and add it to the cache:

```tsx
import { useQueryClient } from '@tanstack/react-query'

function Navigation() {
  const queryClient = useQueryClient()

  const prefetchDashboard = () => {
    queryClient.prefetchQuery({
      queryKey: ['dashboard'],
      queryFn: fetchDashboard,
      staleTime: 60 * 1000, // Cache for 1 minute
    })
  }

  return (
    <nav>
      <a href="/dashboard" onMouseEnter={prefetchDashboard}>
        Dashboard
      </a>
    </nav>
  )
}
```

### usePrefetchQuery (React 19+)

For React 19's prefetching during render:

```tsx
import { usePrefetchQuery } from '@tanstack/react-query'

function ProductCard({ productId }: { productId: string }) {
  // Prefetch during render
  usePrefetchQuery({
    queryKey: ['product', productId],
    queryFn: () => fetchProduct(productId),
  })

  return (
    <a href={`/products/${productId}`}>View Product</a>
  )
}
```

### ensureQueryData

Fetch data only if it's not already cached:

```tsx
import { useQueryClient } from '@tanstack/react-query'

function PostButton({ postId }: { postId: string }) {
  const queryClient = useQueryClient()

  const handleClick = async () => {
    // Fetch only if not in cache
    await queryClient.ensureQueryData({
      queryKey: ['post', postId],
      queryFn: () => fetchPost(postId),
    })

    // Data is guaranteed to be available now
    navigate(`/posts/${postId}`)
  }

  return <button onClick={handleClick}>View Post</button>
}
```

## Prefetching Patterns

### On Hover

Prefetch when user shows intent:

```tsx
import { useQueryClient } from '@tanstack/react-query'

function UserCard({ user }: { user: User }) {
  const queryClient = useQueryClient()

  return (
    <div
      onMouseEnter={() => {
        queryClient.prefetchQuery({
          queryKey: ['user', user.id],
          queryFn: () => fetchUserDetails(user.id),
        })
      }}
    >
      <a href={`/users/${user.id}`}>{user.name}</a>
    </div>
  )
}
```

### On Component Mount

Prefetch related data when component loads:

```tsx
import React from 'react'
import { useQuery, useQueryClient } from '@tanstack/react-query'

function Dashboard() {
  const queryClient = useQueryClient()

  const { data: user } = useQuery({
    queryKey: ['user'],
    queryFn: fetchUser,
  })

  React.useEffect(() => {
    // Prefetch related data
    queryClient.prefetchQuery({
      queryKey: ['notifications'],
      queryFn: fetchNotifications,
    })

    queryClient.prefetchQuery({
      queryKey: ['messages'],
      queryFn: fetchMessages,
    })
  }, [])

  return <DashboardContent user={user} />
}
```

### Prefetch Next Page

Prefetch pagination data:

```tsx
import React from 'react'
import { useQuery, useQueryClient, keepPreviousData } from '@tanstack/react-query'

function PaginatedList() {
  const queryClient = useQueryClient()
  const [page, setPage] = React.useState(1)

  const { data, isPlaceholderData } = useQuery({
    queryKey: ['items', page],
    queryFn: () => fetchItems(page),
    placeholderData: keepPreviousData,
  })

  // Prefetch next page
  React.useEffect(() => {
    if (!isPlaceholderData && data?.hasMore) {
      queryClient.prefetchQuery({
        queryKey: ['items', page + 1],
        queryFn: () => fetchItems(page + 1),
      })
    }
  }, [data, isPlaceholderData, page, queryClient])

  return (
    <div>
      <ItemList items={data?.items} />
      <button onClick={() => setPage(p => p + 1)}>Next</button>
    </div>
  )
}
```

### Prefetch on Route Change

Prefetch before navigation:

```tsx
import { useQueryClient } from '@tanstack/react-query'
import { useNavigate } from 'react-router-dom'

function ProductList() {
  const queryClient = useQueryClient()
  const navigate = useNavigate()

  const goToProduct = async (productId: string) => {
    // Prefetch before navigating
    await queryClient.prefetchQuery({
      queryKey: ['product', productId],
      queryFn: () => fetchProduct(productId),
    })

    // Navigate - data is already cached
    navigate(`/products/${productId}`)
  }

  return (
    <div>
      {products.map(product => (
        <button key={product.id} onClick={() => goToProduct(product.id)}>
          {product.name}
        </button>
      ))}
    </div>
  )
}
```

## Prefetching with Suspense

Prefetch before showing suspense boundary:

```tsx
import React from 'react'
import { useQueryClient } from '@tanstack/react-query'
import { fetchProjects } from './queries'

function Example() {
  const queryClient = useQueryClient()
  const [showProjects, setShowProjects] = React.useState(false)

  return (
    <>
      <button
        onClick={() => {
          setShowProjects((old) => {
            if (!old) {
              // Prefetch before showing
              queryClient.prefetchQuery({
                queryKey: ['projects'],
                queryFn: fetchProjects,
              })
            }
            return !old
          })
        }}
      >
        {showProjects ? 'Hide Projects' : 'Show Projects'}
      </button>

      {showProjects ? (
        <React.Suspense fallback={<h1>Loading projects...</h1>}
          <Projects />
        </React.Suspense>
      ) : null}
    </>
  )
}
```

## Controlling Prefetch Behavior

### With staleTime

Prevent unnecessary prefetches:

```tsx
queryClient.prefetchQuery({
  queryKey: ['user', userId],
  queryFn: () => fetchUser(userId),
  staleTime: 5 * 60 * 1000, // Only prefetch if data is older than 5 minutes
})
```

If data is already cached and fresh, prefetch does nothing.

### Prefetch Multiple Queries

```tsx
import { useQueryClient } from '@tanstack/react-query'

function DashboardLink() {
  const queryClient = useQueryClient()

  const prefetchDashboard = async () => {
    // Prefetch multiple queries in parallel
    await Promise.all([
      queryClient.prefetchQuery({
        queryKey: ['stats'],
        queryFn: fetchStats,
      }),
      queryClient.prefetchQuery({
        queryKey: ['recentActivity'],
        queryFn: fetchRecentActivity,
      }),
      queryClient.prefetchQuery({
        queryKey: ['notifications'],
        queryFn: fetchNotifications,
      }),
    ])
  }

  return (
    <a href="/dashboard" onMouseEnter={prefetchDashboard}>
      Dashboard
    </a>
  )
}
```

## Server-Side Prefetching

### Next.js App Router

```tsx
// app/posts/[id]/page.tsx
import { QueryClient, dehydrate } from '@tanstack/react-query'
import { HydrationBoundary } from '@tanstack/react-query'

export default async function PostPage({
  params,
}: {
  params: { id: string }
}) {
  const queryClient = new QueryClient()

  // Prefetch on server
  await queryClient.prefetchQuery({
    queryKey: ['post', params.id],
    queryFn: () => fetchPost(params.id),
  })

  return (
    <HydrationBoundary state={dehydrate(queryClient)}>
      <Post id={params.id} />
    </HydrationBoundary>
  )
}
```

### Next.js Pages Router

```tsx
import { QueryClient, dehydrate } from '@tanstack/react-query'

export async function getServerSideProps(context) {
  const queryClient = new QueryClient()

  await queryClient.prefetchQuery({
    queryKey: ['post', context.params.id],
    queryFn: () => fetchPost(context.params.id),
  })

  return {
    props: {
      dehydratedState: dehydrate(queryClient),
    },
  }
}
```

## Checking if Data is Cached

Check if data exists before showing UI:

```tsx
import { useQueryClient } from '@tanstack/react-query'

function PostLink({ postId }: { postId: string }) {
  const queryClient = useQueryClient()

  const isCached = queryClient.getQueryData(['post', postId]) !== undefined

  return (
    <a
      href={`/posts/${postId}`}
      style={{ fontWeight: isCached ? 'bold' : 'normal' }}
      onMouseEnter={() => {
        queryClient.prefetchQuery({
          queryKey: ['post', postId],
          queryFn: () => fetchPost(postId),
        })
      }}
    >
      Post {postId}
    </a>
  )
}
```

<Warning>
  **Performance Tip**: Be strategic with prefetching. Prefetching every possible navigation can waste bandwidth and server resources. Focus on high-probability user actions like:
  - Hover on navigation links
  - Next page in pagination
  - Related content visible on screen
  - Common navigation paths
</Warning>

## Common Pitfalls

### Don't Prefetch Everything

```tsx
// ❌ Bad: Prefetches all 100 items immediately
function AllProducts({ products }: { products: Product[] }) {
  const queryClient = useQueryClient()

  React.useEffect(() => {
    products.forEach(product => {
      queryClient.prefetchQuery({
        queryKey: ['product', product.id],
        queryFn: () => fetchProduct(product.id),
      })
    })
  }, [products])
}

// ✅ Good: Prefetch on hover
function AllProducts({ products }: { products: Product[] }) {
  const queryClient = useQueryClient()

  return (
    <div>
      {products.map(product => (
        <div
          key={product.id}
          onMouseEnter={() => {
            queryClient.prefetchQuery({
              queryKey: ['product', product.id],
              queryFn: () => fetchProduct(product.id),
            })
          }}
        >
          {product.name}
        </div>
      ))}
    </div>
  )
}
```
