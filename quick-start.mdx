This guide will walk you through the three core concepts of TanStack Query:

1. **Queries** - Fetching and caching data
2. **Mutations** - Creating, updating, or deleting data
3. **Query Invalidation** - Refetching data after mutations

## Setup the QueryClient

First, you need to create a `QueryClient` and provide it to your application.

## Working Example

Here's a complete example showing queries, mutations, and query invalidation working together:

<CodeGroup>
```tsx React
import {
  useQuery,
  useMutation,
  useQueryClient,
  QueryClient,
  QueryClientProvider,
} from '@tanstack/react-query'
import { getTodos, postTodo } from '../my-api'

// Create a client
const queryClient = new QueryClient()

function App() {
  return (
    // Provide the client to your App
    <QueryClientProvider client={queryClient}>
      <Todos />
    </QueryClientProvider>
  )
}

function Todos() {
  // Access the client
  const queryClient = useQueryClient()

  // Queries
  const query = useQuery({ queryKey: ['todos'], queryFn: getTodos })

  // Mutations
  const mutation = useMutation({
    mutationFn: postTodo,
    onSuccess: () => {
      // Invalidate and refetch
      queryClient.invalidateQueries({ queryKey: ['todos'] })
    },
  })

  return (
    <div>
      <ul>
        {query.data?.map((todo) => (
          <li key={todo.id}>{todo.title}</li>
        ))}
      </ul>

      <button
        onClick={() => {
          mutation.mutate({
            id: Date.now(),
            title: 'Do Laundry',
          })
        }}
      >
        Add Todo
      </button>
    </div>
  )
}
```

```vue Vue
<script setup>
import { useQueryClient, useQuery, useMutation } from '@tanstack/vue-query'
import { getTodos, postTodo } from '../my-api'

// Access QueryClient instance
const queryClient = useQueryClient()

  })

  <span v-if="isPending">Loading...</span>
  <span v-else-if="isError">Error: {{ error.message }}</span>
  <ul v-else>
    <li v-for="todo in data" :key="todo.id">{{ todo.title }}</li>

```

```tsx Solid
import { QueryClient, QueryClientProvider, createQuery, createMutation, useQueryClient } from '@tanstack/solid-query'
import { For } from 'solid-js'
import { getTodos, postTodo } from '../my-api'

const queryClient = new QueryClient()

function Todos() {
  const queryClient = useQueryClient()

  // Query
  const todosQuery = createQuery(() => ({
    queryKey: ['todos'],
    queryFn: getTodos,
  }))

  // Mutation
  const addTodoMutation = createMutation(() => ({
    mutationFn: postTodo,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['todos'] })
    },
  }))

  return (
    <div>
      <ul>
        <For each={todosQuery.data}>
          {(todo) => <li>{todo.title}</li>}
        </For>
      </ul>

      <button
        onClick={() => {
          addTodoMutation.mutate({
            id: Date.now(),
            title: 'Do Laundry',
          })
        }}
      >
        Add Todo
      </button>
    </div>
  )
}

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <Todos />
    </QueryClientProvider>
  )
}
```

```typescript Angular
import { Component, Injectable, inject } from '@angular/core'
import { HttpClient } from '@angular/common/http'
import { lastValueFrom } from 'rxjs'
import {
  injectMutation,
  injectQuery,
  QueryClient,
} from '@tanstack/angular-query-experimental'

interface Todo {
  id: number
  title: string
}

@Injectable({ providedIn: 'root' })
export class TodoService {
  private http = inject(HttpClient)

  getTodos(): Promise<Todo[]> {
    return lastValueFrom(
      this.http.get<Todo[]>('https://jsonplaceholder.typicode.com/todos')
    )
  }

  addTodo(todo: Todo): Promise<Todo> {
    return lastValueFrom(
      this.http.post<Todo>('https://jsonplaceholder.typicode.com/todos', todo)
    )
  }
}

@Component({
  selector: 'app-todos',
  template: `
    <div>
      <ul>
        @for (todo of query.data(); track todo.id) {
          <li>{{ todo.title }}</li>
        }
      </ul>
      
      <button (click)="onAddTodo()">Add Todo</button>
    </div>
  `,
})
export class TodosComponent {
  todoService = inject(TodoService)
  queryClient = inject(QueryClient)

  // Query
  query = injectQuery(() => ({
    queryKey: ['todos'],
    queryFn: () => this.todoService.getTodos(),
  }))

  // Mutation
  mutation = injectMutation(() => ({
    mutationFn: (todo: Todo) => this.todoService.addTodo(todo),
    onSuccess: () => {
      // Invalidate and refetch
      this.queryClient.invalidateQueries({ queryKey: ['todos'] })
    },
  }))

  onAddTodo() {
    this.mutation.mutate({
      id: Date.now(),
      title: 'Do Laundry',
    })
  }
}
```
</CodeGroup>

## Understanding the Core Concepts

### 1. Queries

Queries are used to fetch data. They require:

- **queryKey** - A unique identifier for the query (used for caching)
- **queryFn** - A function that returns a promise

```typescript
const query = useQuery({
  queryKey: ['todos'],
  queryFn: getTodos,
})
```

The query object provides useful properties:

- `data` - The result from the query function
- `isPending` - True if the query is loading for the first time
- `isError` - True if the query encountered an error
- `error` - The error object if `isError` is true
- `isFetching` - True whenever the query is fetching (including background refetches)

<Tip>
Queries automatically refetch in the background when the window regains focus or when the network reconnects. This keeps your data fresh without any extra code!
</Tip>

### 2. Mutations

Mutations are used to create, update, or delete data. They require:

- **mutationFn** - A function that performs the mutation

```typescript
const mutation = useMutation({
  mutationFn: postTodo,
  onSuccess: () => {
    // Handle successful mutation
  },
})
```

Call `mutation.mutate()` to execute the mutation:

```typescript
mutation.mutate({ id: 1, title: 'New Todo' })
```

### 3. Query Invalidation

After a mutation, you typically want to refetch related queries. Use `invalidateQueries` to mark queries as stale:

```typescript
queryClient.invalidateQueries({ queryKey: ['todos'] })
```

This will:
1. Mark all queries with the key `['todos']` as stale
2. Automatically refetch any active queries
3. Refetch on next use for inactive queries

<Note>
Query invalidation is smart - it only refetches queries that are currently being used. Inactive queries are marked as stale and refetch when they become active again.
</Note>

## Real-World Example

Here's a more complete example fetching GitHub repository data:

<CodeGroup>
```tsx React
import { QueryClient, QueryClientProvider, useQuery } from '@tanstack/react-query'
import { ReactQueryDevtools } from '@tanstack/react-query-devtools'

const queryClient = new QueryClient()

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <ReactQueryDevtools />
      <Example />
    </QueryClientProvider>
  )
}

function Example() {
  const { isPending, error, data, isFetching } = useQuery({
    queryKey: ['repoData'],
    queryFn: async () => {
      const response = await fetch(
        'https://api.github.com/repos/TanStack/query',
      )
      return await response.json()
    },
  })

  if (isPending) return 'Loading...'
  if (error) return 'An error has occurred: ' + error.message

  return (
    <div>
      <h1>{data.full_name}</h1>
      <p>{data.description}</p>
      <strong>👀 {data.subscribers_count}</strong>{' '}
      <strong>✨ {data.stargazers_count}</strong>{' '}
      <strong>🍴 {data.forks_count}</strong>
      <div>{isFetching ? 'Updating...' : ''}</div>
    </div>
  )
}
```

```tsx Solid
import { QueryClient, QueryClientProvider, createQuery } from '@tanstack/solid-query'
import { Switch, Match } from 'solid-js'

const queryClient = new QueryClient()

function Example() {
  const query = createQuery(() => ({
    queryKey: ['repoData'],
    queryFn: async () => {
      const response = await fetch(
        'https://api.github.com/repos/TanStack/query',
      )
      return await response.json()
    },
  }))

  return (
    <Switch>
      <Match when={query.isPending}>Loading...</Match>
      <Match when={query.isError}>Error: {query.error.message}</Match>
      <Match when={query.isSuccess}>
        <div>
          <h1>{query.data.full_name}</h1>
          <p>{query.data.description}</p>
          <strong>👀 {query.data.subscribers_count}</strong>{' '}
          <strong>✨ {query.data.stargazers_count}</strong>{' '}
          <strong>🍴 {query.data.forks_count}</strong>
          <div>{query.isFetching ? 'Updating...' : ''}</div>
        </div>
      </Match>
    </Switch>
  )
}

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <Example />
    </QueryClientProvider>
  )
}
```
